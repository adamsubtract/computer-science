function squareRoot(array) {
    var arrayTotal = array.reduce(function(total, amount){
                                  return total + amount;});

    var arraySquared = Math.sqrt(arrayTotal);

    if((arraySquared % 1) === 0 ){
      return arraySquared;
    }
    else return 'does not have a square root';

};
